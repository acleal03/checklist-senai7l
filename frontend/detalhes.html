const api = "http://localhost:3000"; async function login() { const username = document.getElementById("username").value; const senha = document.getElementById("senha").value; if(!username || !senha){ alert("Informe usuário e senha"); return; } try{ const res = await fetch(`${api}/auth/login`,{ method:"POST", headers:{ "Content-Type":"application/json" }, body:JSON.stringify({ username:username, senha:senha }) }); if(!res.ok){ alert("Usuário ou senha inválidos"); return; } const user = await res.json(); localStorage.setItem("usuario",JSON.stringify(user)); window.location.href="dashboard.html"; }catch(e){ console.error(e); alert("Erro ao conectar com o servidor"); } }
